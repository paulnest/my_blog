calcul_nb_editions_int <- function(depuis) {
  annees_debut <- as.integer(str_extract(depuis, "[0-9]{4}"))
  nb_editions <- ceiling((2020 - annees_debut + 1) / 4)
  return(nb_editions)
}

calcul_nb_editions_str <- function(depuis) {
  annees_separees <- str_split(depuis, ", ")[[1]]
  annees_separees <- map_int(annees_separees, ~as.integer(str_extract(., "\\d+")))
  annees_separees <- annees_separees[!is.na(annees_separees) & annees_separees <= 2020]
  if (length(annees_separees) == 0) {
    return(0)  # Aucune année valide trouvée
  }
  nb_editions <- ceiling((2020 - max(annees_separees)) / 4) + 1
  return(nb_editions)
}
