compter_sites_paralympiques <- function(dataframe) {
  nb_site_paralympique <- sum(grepl("Site paralympique", dataframe$sites_olympiques_paralympiques))
  return(list(nb_site_paralympique = nb_site_paralympique))
}

resultat <- compter_sites_paralympiques(jo_paris)

  1. A vol d’oiseau, quels sont les sites les deux sites les plus proches ?
library(geosphere)
library(leaflet)
coordinates <- strsplit(jo_paris$geo_point, ", ")

latitude <- as.numeric(sapply(coordinates, function(x) as.numeric(x[1])))
longitude <- as.numeric(sapply(coordinates, function(x) as.numeric(x[2])))

mymap <- leaflet() %>%
  addTiles() %>%
  addMarkers(lng = longitude, lat = latitude, popup = jo_paris$nom)

mymap
min_distance <- Inf
closest_sites <- c("", "")
for (i in 1:(length(latitude) - 1)) {
  for (j in (i + 1):length(latitude)) {
    distance <- geosphere::distHaversine(c(longitude[i], latitude[i]), c(longitude[j], latitude[j]))
    if (distance < min_distance) {
      min_distance <- distance
      closest_sites <- c(jo_paris$nom[i], jo_paris$nom[j])
    }
  }
}

print(closest_sites)
